- 方案一：双指针
```PYTHON
# Definition for singly-linked list.
# class ListNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution(object):
    def hasCycle(self, head):
        """
        :type head: ListNode
        :rtype: bool
        """
        if  head==None:
            return False
        p = head
        q= head

        while q:
            if q.next==None:
                return False
            else:
                p = p.next
                q = q.next.next
            if p==q:
                return True
        return False
```
